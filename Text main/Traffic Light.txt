		_delay_ms(1000);
		SSD_void_DISPLAY_NUMBER(DIO_PORTB , CATHOD_1);
		_delay_ms(1000);
		SSD_void_DISPLAY_NUMBER(DIO_PORTB , CATHOD_0);
		_delay_ms(1000);

		LED_voidLED_OFF(DIO_PORTA , DIO_PIN0);
		LED_voidLED_ON(DIO_PORTA , DIO_PIN1);

		SSD_void_DISPLAY_NUMBER(DIO_PORTB , CATHOD_3);

		_delay_ms(1000);

		LED_voidLED_OFF(DIO_PORTA , DIO_PIN1);
		LED_voidLED_ON(DIO_PORTA , DIO_PIN2);

		SSD_void_DISPLAY_NUMBER(DIO_PORTB , CATHOD_9);
		_delay_ms(1000);
		SSD_void_DISPLAY_NUMBER(DIO_PORTB , CATHOD_8);
		_delay_ms(1000);
		SSD_void_DISPLAY_NUMBER(DIO_PORTB , CATHOD_7);
		_delay_ms(1000);
